class Solution:
    def countGood(self, nums: List[int], k: int) -> int:
        
        left = 0
        pairs = 0
        count = 0
        windows = {}

        for right in range(len(nums)):
            num = nums[right]
            pairs  += windows.get(num, 0)
            windows[num] = windows.get(num, 0) + 1

            while pairs >= k:
                windows[nums[left]] -= 1
                count += len(nums) - right
                pairs -= windows[nums[left]]
                left += 1
        
        return count
