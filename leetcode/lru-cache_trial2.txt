class Node:
    def __init__(self, key, value):
        self.key = key
        self.val = value
        self.prev = None
        self.next = None

class LRUCache:

    def __init__(self, capacity: int):
        self.cap = capacity
        self.catch = {}
        self.left = Node(0,0)
        self.right = Node(0,0)
        self.left.next = self.right
        self.right.prev = self.left

    def remove(self,node):
        prv = node.prev
        nxt = node.next
        prv.next = nxt
        nxt.prev = prv
    def insert(self, node):
        prv = self.right.prev
        prv.next = node
        node.prev = prv
        node.next = self.right
        self.right.prev = node

    def get(self, key: int) -> int:
        if key in self.catch :
            node = self.catch[key]
            self.remove(node) # remove from where it is and
            self.insert(node) # make it most resent
            return node.val
        return - 1

    def put(self, key: int, value: int) -> None:
        if key in self.catch:
            self.remove(self.catch[key])
        node = Node(key, value)
        self.catch[key] = node
        self.insert(node)

        if len(self.catch) > self.cap:
            lru = self.left.next
            self.remove(lru)
            del self.catch[lru.key]
