class Solution:
    def judgeSquareSum(self, c: int) -> bool:
        left = 0
        right = int(math.sqrt(c))

        while left <= right:
            a = left*left
            b = right*right
            if a + b == c:
                return True
            elif  a + b < c:
                left += 1
            else:
                right -= 1
        return False
