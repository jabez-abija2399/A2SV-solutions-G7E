class Solution:
    def pancakeSort(self, arr: List[int]) -> List[int]:
        result = []
        for size in range(len(arr), 1, -1):
            maxIndex = arr.index(max(arr[:size]))

            if maxIndex == size - 1:
                continue
            if maxIndex != 0:
                result.append(maxIndex + 1)
                arr[:maxIndex + 1] = reversed(arr[:maxIndex + 1])
            result.append(size)
            arr[:size] = reversed(arr[:size])
        return result
