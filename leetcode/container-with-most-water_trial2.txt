class Solution:
    def maxArea(self, height: List[int]) -> int:
        n = len(height)
        right = n -1
        left = 0
        maxWater = 0
        while left < right:
            width = right - left
            hght = min(height[left], height[right])
            maxWater = max(maxWater, width * hght)
            if height[left] < height[right]:
                left += 1
            else:
                right -= 1
        return maxWater
