class Solution:
    def maxDistance(self, position: List[int], m: int) -> int:
        position.sort()
        def canPlace(d):
            count = 1
            lastPos = position[0]

            for i in range(1, len(position)):
                if position[i] - lastPos >= d:
                    count += 1
                    lastPos = position[i]
                    if count == m:
                        return True
            return False

        l, r = 1, position[-1] - position[0]
        res = 0

        while l <= r:
            mid = (l + r) // 2

            if canPlace(mid):
                res = mid
                l = mid + 1
            else:
                r = mid - 1
        return res
