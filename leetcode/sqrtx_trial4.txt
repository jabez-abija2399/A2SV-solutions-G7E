class Solution:
    def mySqrt(self, x: int) -> int:
        l = 0 # 0 0 3
        r = x # 9 3 
        ans = 0 # 3
        # 1 2 3 4 5 6 7 8 9
        while l <= r:
            mid = (l+r) // 2 # 4 2
            if mid * mid <= x:
                ans = mid
                l = mid + 1
                
            else:
                r = mid - 1
        return ans
