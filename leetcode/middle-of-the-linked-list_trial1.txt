# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def middleNode(self, head: Optional[ListNode]) -> Optional[ListNode]:
        cur = head
        leng = 0
        while cur:
            leng += 1
            cur = cur.next
        cur = head
        for _ in range(leng // 2):
            cur = cur.next
        return cur
