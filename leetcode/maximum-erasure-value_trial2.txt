class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        left  =0 
        maxSum = 0
        seen = set()
        winSum = 0
        for right in range(len(nums)):
            num = nums[right]
            
            while num in seen:
                seen.remove(nums[left])
                winSum -= nums[left]
                left += 1
            seen.add(num)
            winSum += num

            maxSum = max(maxSum, winSum)
        return maxSum
