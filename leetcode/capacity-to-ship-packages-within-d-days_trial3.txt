class Solution:
    def shipWithinDays(self, weights: List[int], days: int) -> int:
        l = max(weights)
        r = sum(weights)

        while l < r:
            mid = (l + r) // 2

            if self.isShipped(weights, mid, days):
                r = mid
            else:
                l = mid + 1
        return r

    def isShipped(self, weights, mid, days):
        curWeight = 0
        dayTaken = 1

        for weight in weights:
            curWeight += weight

            if curWeight > mid:
                dayTaken += 1
                curWeight = weight
        return dayTaken <= days
