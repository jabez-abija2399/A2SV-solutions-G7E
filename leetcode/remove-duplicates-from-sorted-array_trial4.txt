class Solution:
    def removeDuplicates(self, nums: List[int]) -> int:
        k = 1

        
        for i in range(1,len(nums)):
            if nums[k-1] != nums[i]:

                nums[k] = nums[i]
                k += 1

            
        return k
        # 0,0,1,1,1,2,2,3
       
        """
         i = 0, 1,  2,3,4,  5,6
        k = 1,2 3
        0 1 1 1 1 2 2 3
        0 1 2 1 1 2 2 3
        """
