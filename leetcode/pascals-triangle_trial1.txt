class Solution:
    def generate(self, numRows: int) -> List[List[int]]:
        if numRows == 1:
            return [[1]]
        tria = self.generate(numRows -1)
        prevRow = tria[-1]
        newRow = [1]

        for i in range(1, len(prevRow)):
            newRow.append(prevRow[i-1] + prevRow[i])
        newRow.append(1)
        tria.append(newRow)

        return tria
